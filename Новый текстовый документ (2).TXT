#include <iostream>
#include <locale.h>

using namespace std;

int main() {
    // Настройка для русского языка
    setlocale(LC_ALL, "Russian");
    
    int number;
    int i = 1; // Счетчик от 1 до 10
    
    cout << "[+] Введите число: ";
    cin >> number;
    
    cout << "\n";
    
    // Выводим таблицу умножения с помощью цикла while
    while (i <= 10) {
        cout << "[+] " << number << " × " << i << " = " << number * i << endl;
        i = i + 1; // Увеличиваем счетчик на 1
    }
    
    return 0;
}
