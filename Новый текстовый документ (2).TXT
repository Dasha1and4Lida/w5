#include <iostream>
#include <locale.h>

using namespace std;

int main() {
    setlocale(LC_ALL, "Russian");
    
    int number;
    int i = 1;
    
    cout << "[+] Введите число: ";
    cin >> number;
    
    cout << "\n";
    
    
    while (i <= 10) {
        cout << "[+] " << number << " × " << i << " = " << number * i << endl;
        i = i + 1;
    }
    
    return 0;

}
